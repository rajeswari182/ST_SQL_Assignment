1. Countries and their continents. Order the results by continent names.
    
SELECT 
    co.continent, 
    c.countryname
FROM 
    countries c
JOIN 
    continents co ON co.contid = c.continent
ORDER BY 
    c.countryname;


---


2. Number of countries within each continent
    
 SELECT 
    co.continent, 
    COUNT(c.countryname) AS total_countries
FROM 
    countries c
JOIN 
    continents co ON co.contid = c.continent
GROUP BY 
    co.continent
HAVING 
    COUNT(c.countryname) >= 2
ORDER BY 
    total_countries DESC;

3. countries with no car makers?

 b. Cylinders
 c. Edispl
 d. Horsepower
 e. Weight
 f. Acel
    
A.MPG-

SELECT md.Model, cm.Maker, cm.FullName, c.CountryName, co.Continent, cd.mpg
FROM Car_Details cd
JOIN Car_Names cn ON cd.ID = cn.ID
JOIN Model_Details md ON cn.Model = md.Model
JOIN Car_Makers cm ON md.Maker = cm.ID
JOIN Countries c ON cm.Country = c.CountryID

11. In which year most cars were made?
    
SELECT 
    year, 
    COUNT(id) AS total_cars
FROM 
    car_details
GROUP BY 
    year
ORDER BY 
    total_cars DESC
OFFSET 1 ROWS FETCH NEXT 1 ROW ONLY;


-----


    
12. Which year had less cars made?
    
SELECT 
    year, 
    COUNT(id) AS total_cars
FROM 
    car_details
GROUP BY 
    year
ORDER BY 
    total_cars ASC
OFFSET 1 ROWS FETCH NEXT 1 ROW ONLY;
